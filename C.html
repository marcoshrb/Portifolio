float rendimento(float montante, float valor) { return montante - valor; }
float tributacao (float rendimento, int tempo)
{
if (tempo <= 3)
rendimento = rendimento*0.775;
if (tempo > 3 && tempo <= 12)
rendimento = rendimento*0.80;
if (tempo > 12 && tempo <= 24)
rendimento = rendimento*0.825;
if (tempo > 24)
rendimento = rendimento*0.85;
return rendimento;
}
